# Binaries are NOT in this repo (licensing). They live in ~/Keys/fonts/.
#
# Restore on a fresh machine:
#   cp ~/Keys/fonts/*.{otf,ttf} ~/Library/Fonts/
# Then re-launch any app that has the font selected so it picks up the new family.
