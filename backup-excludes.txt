# rsync exclude patterns for backing up $HOME to an external drive.
# Usage (or just run: ~/dotfiles/scripts/backup-home.sh /Volumes/BackupSSD/home):
#   rsync -aHv --delete --exclude-from=~/dotfiles/backup-excludes.txt \
#         ~/ /Volumes/BackupSSD/home/
#
# Note: rsync does NOT strip trailing comments after a pattern, so all
# comments must be on their own lines.

# ─── Top-level skips (anchored to $HOME root) ───

# All hidden dot-entries at $HOME root: .config, .cache, .aws, .nvm, .docker,
# .ssh, etc. Real configs live in ~/dotfiles; secrets in ~/Keys; the rest is
# rebuildable. (Hidden dirs nested inside backed-up folders, like project
# .git/ dirs, are NOT affected — only the leading-slash anchors to root.)
/.*

# Custom user binaries — keep them in dotfiles or reinstall.
/bin/

# User-managed packages, often large + reinstallable.
/opt/

# User-installed apps — covered by Brewfile + manual-apps.txt.
/Applications/

# Massive, mostly caches/DBs; specific app exports handled separately.
/Library/

# Browser overrides — rebuildable.
/overrides_chrome/
